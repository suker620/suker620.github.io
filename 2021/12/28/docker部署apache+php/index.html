docker部署apache+php

#1、拉取镜像
docker pull php:7.0-apache
#2、创建src目录，存放web目录
mkdir /data/src
#3、创建测试文件
vim /data/src/info.php

#4、创建容器
docker run -itd -v /data/src:/var/www/html -p 80:80 php:7.0-apache
#5、测试访问
http://192.168.3.5/info.php
